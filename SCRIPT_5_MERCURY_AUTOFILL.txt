// ==UserScript==
// @name         Mercury Auto Fill & Verify Assistant
// @namespace    http://tampermonkey.net/
// @version      1.0
// @description  Tu dong dien thong tin dang ky tai khoan doanh nghiep Mercury Bank tu dong bo local
// @author       Antigravity AI
// @match        https://dashboard.mercury.co/*
// @grant        GM_xmlhttpRequest
// @grant        GM_setValue
// @grant        GM_getValue
// @connect      127.0.0.1
// @connect      localhost
// @run-at       document-end
// ==/UserScript==

(function() {
    'use strict';

    let debugDiv = null;
    function initDebug() {
        if (debugDiv) return;
        debugDiv = document.createElement('div');
        debugDiv.style.position = 'fixed';
        debugDiv.style.top = '10px';
        debugDiv.style.right = '10px';
        debugDiv.style.background = '#1e1b4b';
        debugDiv.style.color = 'white';
        debugDiv.style.padding = '12px 16px';
        debugDiv.style.borderRadius = '8px';
        debugDiv.style.boxShadow = '0 10px 15px -3px rgba(0, 0, 0, 0.3)';
        debugDiv.style.zIndex = '9999999';
        debugDiv.style.fontSize = '13px';
        debugDiv.style.fontFamily = 'monospace';
        debugDiv.style.lineHeight = '1.5';
        debugDiv.style.maxWidth = '400px';
        debugDiv.style.wordBreak = 'break-all';
        debugDiv.innerHTML = '⚖️ [Mercury Assistant] Dang khoi dong...';
        document.body.appendChild(debugDiv);
    }

    function setDebug(msg, color = '#1e1b4b') {
        if (!debugDiv) initDebug();
        debugDiv.style.background = color;
        debugDiv.innerHTML = '⚖️ [Mercury Assistant]<br>' + msg;
    }

    const API_CURRENT_ACCOUNT = 'http://127.0.0.1:3000/api/current-account';

    function safeFill(element, value) {
        if (!element) return false;
        try {
            element.focus();
            const prototype = Object.getPrototypeOf(element);
            const descriptor = Object.getOwnPropertyDescriptor(prototype, "value");
            
            if (descriptor && descriptor.set) {
                descriptor.set.call(element, value);
            } else {
                element.value = value;
            }
            
            element.dispatchEvent(new Event('input', { bubbles: true }));
            element.dispatchEvent(new Event('change', { bubbles: true }));
            element.dispatchEvent(new KeyboardEvent('keydown', { bubbles: true, key: value }));
            element.dispatchEvent(new KeyboardEvent('keyup', { bubbles: true, key: value }));
        } catch (e) {
            console.error('[Tampermonkey] Loi safeFill:', e);
            element.value = value;
            element.dispatchEvent(new Event('input', { bubbles: true }));
            element.dispatchEvent(new Event('change', { bubbles: true }));
        }
        return true;
    }

    function safeClick(element) {
        if (!element) return false;
        try {
            element.focus();
            const clickEvent = new MouseEvent('click', {
                bubbles: true,
                cancelable: true,
                view: window
            });
            element.dispatchEvent(clickEvent);
            if (typeof element.click === 'function') {
                element.click();
            }
        } catch (e) {
            console.error('[Tampermonkey] Loi safeClick:', e);
            element.click();
        }
        return true;
    }

    function makeRequest(url, method, data, callback) {
        if (typeof GM_xmlhttpRequest !== 'undefined') {
            GM_xmlhttpRequest({
                method: method,
                url: url,
                headers: { "Content-Type": "application/json" },
                data: data ? JSON.stringify(data) : null,
                onload: function(res) { callback(null, res.responseText); },
                onerror: function(err) {
                    fetch(url, {
                        method: method,
                        headers: data ? { "Content-Type": "application/json" } : {},
                        body: data ? JSON.stringify(data) : null
                    })
                    .then(r => r.text())
                    .then(text => callback(null, text))
                    .catch(e => callback(e, null));
                }
            });
        }
    }

    window.addEventListener('DOMContentLoaded', () => {
        initDebug();
        
        const interval = setInterval(() => {
            try {
                makeRequest(API_CURRENT_ACCOUNT, "GET", null, function(err, responseText) {
                    if (err) {
                        setDebug('⏳ Khong ket noi duoc voi Tool local...', '#be123c');
                        return;
                    }
                    try {
                        const resData = JSON.parse(responseText);
                        if (resData.success && resData.account && resData.account.isMercury) {
                            const acc = resData.account;
                            setDebug(`Dong bo OK! 🚀<br>🏢 Doanh nghiep: ${acc.nameOrg}<br>👤 Chu so huu: ${acc.fullName}`, '#16a34a');
                            runStep(acc);
                        } else {
                            setDebug('⏳ Vui long click [Vao Mercury] tren giao dien de truyen thong tin...', '#475569');
                        }
                    } catch (e) {
                        console.error('Loi parse account:', e);
                    }
                });
            } catch (e) {
                setDebug('Loi vong lap: ' + e.message, '#be123c');
            }
        }, 1500);
    });

    function runStep(acc) {
        const path = window.location.pathname;

        // 1. BUOC TẠO TÀI KHOẢN (CREATE ACCOUNT - Email & Password)
        if (path === '/signup' || path.includes('/signup')) {
            const emailInput = document.querySelector('input[type="email"], input[name="email"], input[id*="email"]');
            const passInput = document.querySelector('input[type="password"], input[name="password"], input[id*="password"]');
            
            // Tìm First Name / Last Name nếu có xuất hiện ở bước 1
            const firstNameInput = document.querySelector('input[name="firstName"], input[id="firstName"], input[placeholder*="First name"]');
            const lastNameInput = document.querySelector('input[name="lastName"], input[id="lastName"], input[placeholder*="Last name"]');

            if (firstNameInput && firstNameInput.offsetWidth > 0 && acc.fullName) {
                const parts = acc.fullName.split(' ');
                const firstName = parts[0] || '';
                const lastName = parts.slice(1).join(' ') || '';
                
                if (firstNameInput.value !== firstName) {
                    setDebug(`✍️ Dien First Name: ${firstName}`, '#2563eb');
                    safeFill(firstNameInput, firstName);
                }
                if (lastNameInput && lastNameInput.value !== lastName) {
                    setDebug(`✍️ Dien Last Name: ${lastName}`, '#2563eb');
                    safeFill(lastNameInput, lastName);
                }
            }

            if (emailInput && emailInput.offsetWidth > 0 && acc.email) {
                if (emailInput.value !== acc.email) {
                    setDebug(`✍️ Dien Email dang ky: ${acc.email}`, '#2563eb');
                    safeFill(emailInput, acc.email);
                }
            }

            if (passInput && passInput.offsetWidth > 0 && acc.password) {
                if (passInput.value !== acc.password) {
                    setDebug(`✍️ Dien Mat khau ngau nhien...`, '#2563eb');
                    safeFill(passInput, acc.password);
                }
            }
            return;
        }

        // 2. BUOC DIEN THONG TIN CHI TIET PROFILE NGUOI DUNG (YOUR DETAILS / YOUR PROFILE)
        const isDetailsPage = document.querySelector('h1, h2, div')?.textContent?.toLowerCase().includes('your details') ||
                              document.querySelector('h1, h2, div')?.textContent?.toLowerCase().includes('your profile') ||
                              document.querySelector('input[name="ownershipPercentage"]') !== null;

        if (isDetailsPage) {
            // Tỷ lệ sở hữu (Ownership percentage)
            const ownershipInput = document.querySelector('input[name="ownershipPercentage"], input[placeholder*="Ownership"]');
            if (ownershipInput && ownershipInput.offsetWidth > 0 && ownershipInput.value !== '100') {
                setDebug('✍️ Dien ty le so huu: 100%', '#2563eb');
                safeFill(ownershipInput, '100');
            }

            // Ngày sinh (Birthday Month, Day, Year)
            const dobInput = acc.dob || ''; // Dạng "Oct 20 1987"
            if (dobInput) {
                const parts = dobInput.split(' ');
                if (parts.length === 3) {
                    const months = { 'jan': '01', 'feb': '02', 'mar': '03', 'apr': '04', 'may': '05', 'jun': '06', 'jul': '07', 'aug': '08', 'sep': '09', 'oct': '10', 'nov': '11', 'dec': '12' };
                    const monthText = parts[0].toLowerCase().slice(0, 3);
                    const monthNum = months[monthText] || '10';
                    const dayNum = parts[1].replace(/[^0-9]/g, '').padStart(2, '0');
                    const yearNum = parts[2].trim();

                    const monthInput = document.querySelector('input[placeholder="MM"], input[name*="month"], input[id*="month"]');
                    const dayInput = document.querySelector('input[placeholder="DD"], input[name*="day"], input[id*="day"]');
                    const yearInput = document.querySelector('input[placeholder="YYYY"], input[name*="year"], input[id*="year"]');

                    if (monthInput && monthInput.value !== monthNum) {
                        setDebug(`✍️ Dien ngay sinh: ${monthNum}/${dayNum}/${yearNum}`, '#2563eb');
                        safeFill(monthInput, monthNum);
                    }
                    if (dayInput && dayInput.value !== dayNum) {
                        safeFill(dayInput, dayNum);
                    }
                    if (yearInput && yearInput.value !== yearNum) {
                        safeFill(yearInput, yearNum);
                    }
                }
            }

            // Số an sinh xã hội (SSN)
            const ssnInput = document.querySelector('input[name="ssn"], input[id*="ssn"], input[placeholder*="SSN"]');
            if (ssnInput && ssnInput.offsetWidth > 0 && ssnInput.value !== acc.ssn) {
                setDebug(`✍️ Dien SSN: ${acc.ssn}`, '#2563eb');
                safeFill(ssnInput, acc.ssn);
            }

            // Địa chỉ nhà riêng (Residential Address)
            const addressInput = document.querySelector('input[placeholder*="Search for an address"], input[name*="address"], input[id*="address"]');
            const fullPersonalAddress = `${acc.personalAddress}, ${acc.city}, ${acc.personalState} ${acc.zipcode}`;
            if (addressInput && addressInput.offsetWidth > 0 && addressInput.value.length < 5) {
                setDebug(`✍️ Dien Dia chi rieng: ${fullPersonalAddress}`, '#2563eb');
                safeFill(addressInput, fullPersonalAddress);
            }
            return;
        }

        // 3. BUOC DIEN THONG TIN DOANH NGHIEP (BUSINESS DETAILS)
        const isBusinessPage = document.querySelector('h1, h2, div')?.textContent?.toLowerCase().includes('business details') ||
                               document.querySelector('input[name="ein"]') !== null;

        if (isBusinessPage) {
            // Tên doanh nghiệp (Legal Business Name)
            const orgInput = document.querySelector('input[name="legalBusinessName"], input[placeholder*="Legal name"]');
            if (orgInput && orgInput.offsetWidth > 0 && orgInput.value !== acc.nameOrg) {
                setDebug(`✍️ Dien ten cong ty: ${acc.nameOrg}`, '#2563eb');
                safeFill(orgInput, acc.nameOrg);
            }

            // Mã số thuế doanh nghiệp (EIN)
            const einInput = document.querySelector('input[name="ein"], input[id*="ein"], input[placeholder*="EIN"]');
            if (einInput && einInput.offsetWidth > 0 && einInput.value !== acc.ein) {
                setDebug(`✍️ Dien EIN: ${acc.ein}`, '#2563eb');
                safeFill(einInput, acc.ein);
            }

            // Địa chỉ doanh nghiệp (Business Address)
            const bizAddressInput = document.querySelector('input[placeholder*="Search for an address"], input[name*="address"], input[id*="address"]');
            if (bizAddressInput && bizAddressInput.offsetWidth > 0 && bizAddressInput.value.length < 5) {
                setDebug(`✍️ Dien Dia chi cong ty: ${acc.businessAddress}`, '#2563eb');
                safeFill(bizAddressInput, acc.businessAddress);
            }
            return;
        }
    }
})();
